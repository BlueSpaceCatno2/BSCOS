cmake_minimum_required(VERSION 3.10)

set(CMAKE_ASM_NASM_COMPILER "/usr/bin/nasm" CACHE FILEPATH "Path to NASM")
enable_language(C)
enable_language(ASM_NASM)
project(BSCOS C ASM_NASM)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -m32 -ffreestanding -fno-stack-protector -fno-pic -nostdlib")
set(CMAKE_ASM_NASM_FLAGS "-f elf32")
add_executable(bscos.bin
        boot.s
        main.c
)
target_link_options(bscos.bin PRIVATE
        -melf_i386
        -T ${CMAKE_SOURCE_DIR}/linker.ld
)
add_custom_command(TARGET bscos.bin POST_BUILD
        COMMAND mkdir -p isodir/boot/grub
        COMMAND cp bscos.bin isodir/boot/bscos.bin
        COMMAND echo "menuentry 'BSCOS' { multiboot /boot/bscos.bin }" > isodir/boot/grub/grub.cfg
        COMMAND grub-mkrescue -o bscos.iso isodir
        COMMENT "--- Universal ISO Created: bscos.iso ---"
)
